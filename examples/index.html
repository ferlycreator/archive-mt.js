添加点

通过new Z.Marker(...)创建一个点，并添加到图层上。

//创建点的坐标对象
var coordinate = new Z.Coordinate(121.489935,31.24432);
//创建点对象
var marker = new Z.Marker(coordinate);
//将点添加到Layer
layer.addGeometry(marker);

标注图片

1、通过new Z.Marker(...)创建一个点；

2、构造图片参数，并设置到Marker；

3、将点添加到图层上。

//创建点的坐标对象
var coordinate = new Z.Coordinate(121.489935,31.24432);
//设置图片
var icon = {
    'marker-file': '/engine/images/marker.png',
    'marker-width': 22,
    'marker-height': 30,
    'dx': 0,
    'dy' :0
};
//创建点对象
var marker = new Z.Marker(coordinate, 'marker_id');
//设置点样式
marker.setSymbol(icon);
//将点添加到Layer
layer.addGeometry(marker);

标注矢量图形

1、通过new Z.Marker(...)创建一个点；

2、构造矢量图形参数，并设置到Marker；

3、将点添加到图层上。

//创建点的坐标对象
var coordinate = new Z.Coordinate(121.489935,31.24432);
//设置矢量配置
var icon = {
    'marker-type': 'triangle',
    'marker-opacity': 1,
    'marker-fill-opacity': 1,
    'marker-line-color': '#0000ff',
    'marker-line-width': 1,
    'marker-line-opacity': 1,
    'marker-placement': 'point',
    'marker-width': 30,
    'marker-height': 30,
    'marker-fill': '#ff0000',
    'dx': 0,
    'dy': 0
};
/**
* marker-type 说明：
* circle：圆形；triangle：三角形；cross：十字；diamond：菱形；square：正方形；x/X：X形；bar：柱状图
*/
//创建点对象
var marker = new Z.Marker(coordinate, 'marker_id');
//设置点样式
marker.setSymbol(icon);
//将点添加到Layer
layer.addGeometry(marker);

设置点标签

1、通过new Z.Marker(...)创建一个点；

2、构造标签参数，并设置到Marker；

3、将点添加到图层上。

//创建点的坐标对象
var coordinate = new Z.Coordinate(121.489935,31.24432);
//设置矢量配置
var icon = {
    'shield-type': 'tip',//label tip
    'shield-name': '测试标签: [marker_name]',
    'shield-opacity': 1,
    'shield-line-color': '#121212',
    'shield-line-width': 1,
    'shield-line-opacity': 1,
    'shield-fill': '#536457',
    'shield-fill-opacity': 1,
    'shield-file': '',
    'shield-face-name': 'Serif',
    'shield-unlock-image' : false,
    'shield-size': 12,
    'shield-text-fill': '#ff0000',
    'shield-placement': 'point', //point line vertex interior
    'shield-wrap-width': 100,
    'shield-wrap-before': false,
    'shield-wrap-character': '',
    'shield-character-spacing': 0,
    'shield-line-spacing': 8,
    'shield-text-dx': 0,
    'shield-text-dy': 0,
    'shield-dx': 0,
    'shield-dy': 0,
    'shield-text-opacity': 1,
    'shield-horizontal-alignment': 'middle',//left middle right
    'shield-vertical-alignment': 'top',//top middle bottom
    'shield-justify-alignment': 'left'//left center right
};
/**
* 说明：
* 1、shield-type tip：气泡形；label：矩形标签；
* 2、在shield-type 为tip的情形下，设置shield-horizontal-alignment 与 shield-vertical-alignment
*    气泡的尖端会根据位置调整。
*/
//创建点对象
var marker = new Z.Marker(coordinate, 'marker_id');
marker.setProperties({
    'marker_name':'上海大厦',
    'marker_addr': '北苏州路20号'
});
//设置点样式
marker.setSymbol(icon);
//将点添加到Layer
layer.addGeometry(marker);
