快速上手 MapTalks

该向导帮助您将快速掌握构建地图、标注点、线与面，并为它们设置标签，以及监听处理交互事件。

第一张定制地图

1、引入 MapTalks JavaScript 文件；

<link href="http://www.maptalks.org/maptalks/v2/css/maptalks.min.js" rel="stylesheet" type="text/css"/>

2、在页面中添加一个放置Map的Html Element，例如：DIV；

<div id="mapObj" style="width:100%; height:100%;"></div>

3、使用 new Z.Map(...)创建地图对象，并设置底图图层。

<!-- 第一步：MapTalks JavaScript 开发包 -->
<script type="text/javascript" src="http://www.maptalks.org/maptalks/v2/css/maptalks.min.js"></script>
<body scroll="no"  onload="initMap();">
    <!-- 第二步、设置放置地图的Html Element -->
    <div id="mapObj" style="width:100%;height:100%;float:right;margin-top:0px"></div>
</body>
<!-- 第三步、创建地图对象，并设置底图图层 -->
<script type="text/javascript" >
    var mapObj;
    function initMap(){
        //创建地图对象
        mapObj = new Z.Map('mapObj', //放置mapObj的div id
        {
            center:  new Z.Coordinate(121.489935,31.24432), //初始中心点坐标
            zoomLevel:   11, //初始缩放级别
            attributionControl:  false, //版权信息控件
            zoomControl :  false, //缩放控件
            navControl:  false, //导航控件
            scaleControl:  false //比例尺控件
        });
        //定制地图底图(Tile Layer)
        var tileLayer = new Z.TileLayer('tile', //底图id
        {
            tileInfo: 'web-mercator', //坐标系
            urlTemplate:'http://emap{s}.mapabc.com/mapabc/maptile?&x={x}&y={y}&z={z}',//底图服务url
            subdomains:[1,2,3]//对应urlTemplate中占位符{s}的值
        });
        //设置底图图层
        mapObj.setBaseTileLayer(tileLayer);
    }
</script>

添加点、线与面

1、使用new Z.VectorLayer(...)创建放置图形的图层；

2、使用new Z.Marker(...)创建点；

3、使用new Z.Polyline(...)创建线；

4、使用new Z.Polygon(...)创建面；

5、将点、线与面添加到图层上。

//创建放置图形的图层
var layer = new Z.VectorLayer('layer');
//将图层添加到地图
mapObj.addLayer(layer);

//创建点对象
var marker = new Z.Marker(new Z.Coordinate(121.489935,31.24432));
//设置点图片
var icon = {
    'marker-file': '/maptalks/v2/images/marker.png',
    'marker-width': 22,
    'marker-height': 30,
    'dx': 0,
    'dy' :0
};
//设置点样式
marker.setSymbol(icon);
//将点添加到Layer
layer.addGeometry(marker);

//创建线对象
var polyline = new Z.Polyline(
    //线端点坐标数组
    [new Z.Coordinate(121.454716,31.272095),
    new Z.Coordinate(121.459007,31.25933),
    new Z.Coordinate(121.458149,31.241572),
    new Z.Coordinate(121.460381,31.229977),
    new Z.Coordinate(121.473427,31.222784)]);
//设置线样式
var polylineSymbol = {
    'line-color' : '#ff00ff',
    'line-width' : 5,
    'line-dasharray' : '-',//线形
    'line-opacity' : 1
};
polyline.setSymbol(polylineSymbol);
//将线添加到Layer
layer.addGeometry(polyline);

//创建面对象
var polygon = new Z.Polygon(
    //面端点坐标数组
    [[new Z.Coordinate(121.508617,31.241279),
    new Z.Coordinate(121.519432,31.241572),
    new Z.Coordinate(121.524067,31.241719),
    new Z.Coordinate(121.528015,31.242893),
    new Z.Coordinate(121.539173,31.245388),
    new Z.Coordinate(121.544323,31.236142),
    new Z.Coordinate(121.548443,31.227335),
    new Z.Coordinate(121.538486,31.223518),
    new Z.Coordinate(121.517887,31.223665)]
    ]);
//设置面样式
var polygonSymbol = {
    'line-color' : '#0000ff',
    'line-width' : 2,
    'line-dasharray' : '-',//线形
    'line-opacity' : 1,
    'polygon-fill' : '#ff00ff',
    'polygon-opacity' : 0.8
};
polygon.setSymbol(polygonSymbol);
//将面添加到Layer
layer.addGeometry(polygon);

底图

设置底图

1、使用new Z.TileLayer(...),构造底图图层；

2、为mapObj设置底图图层。

//定制地图底图(Tile Layer)
var tileLayer = new Z.TileLayer('tile', //底图id
{
    /**
    * 坐标系类型：
    * web-mercator：谷歌, 必应,高德, 腾讯等地图服务采用的瓦片系统;
    * baidu: 百度标注。
    */
    tileInfo: 'web-mercator',
    /**
    * 在线底图服务url
    */
    urlTemplate:'http://emap{s}.mapabc.com/mapabc/maptile?&x={x}&y={y}&z={z}',
    /**
    *urlTemplate中占位符{s}所有数值的数组
    */
    subdomains:[1,2,3]
});
//设置底图图层
mapObj.setBaseTileLayer(tileLayer);

图层

添加图层

1、使用new Z.VectorLayer(...)创建一个图层；

2、将图层添加到mapObj。

/**
* 创建放置图形的图层
* 说明：new Z.VectorLayer(id, {'render': 'Canvas'});
* id:图层的名称；render:渲染方式，默认使用SVG或VML（ie9以下），支持5万以内图形渲染
* 如果需要渲染5万以上的图形，建议设置渲染方式为Canvas {'render': 'Canvas'}
* 注意：Canvas渲染只支持ie8以上浏览器，以及chrome与safari、firefox等
* 下面例子就采用Canvas渲染10000个点
*/
var layer = new Z.VectorLayer('layer',{'render': 'Canvas'});
//将图层添加到地图
mapObj.addLayer(layer);

标注点

1、使用new Z.DrawTool(...)创建drawTool；

2、将drawTool添加到需要绘制工具的map上。

//构造绘制点工具
drawTool = new Z.DrawTool({
    mode: Z.Geometry.TYPE_POINT//设置drawTool绘制的图形类型
});
//将点绘制工具绑定到地图对象
drawTool.addTo(mapObj);
//监听绘制工具完成事件，获取绘制的点坐标等信息
drawTool.on('afterdraw', onAfterDrawMarker);
//绘制完点后操作
function onAfterDrawMarker(param) {
    console.log(param);
    var coordinate = param.coordinate;
    alert('您绘制的点的坐标为:{x:'+coordinate.x+',y:'+coordinate.y);        //创建点的坐标对象
    //设置图片
    var icon = {
        'marker-file': '/maptalks/v2/images/marker.png',
        'marker-width': 22,
        'marker-height': 30,
        'dx': 0,
        'dy' :0
    };
    //创建点对象
    var marker = new Z.Marker(coordinate,{'id':'marker_id'});
    //设置点样式
    marker.setSymbol(icon);
    //将点添加到Layer
    layer.addGeometry(marker);
}

地图控件

ZoomIn/ZoomOut控件

1、引入control 样式文件<link href="/maptalks/v2/css/controls.css" rel="stylesheet" type="text/css"/>；

2、配置mapObj属性zoomControl: true/false打开或关闭zoom控件；

3、使用new Z.Control.Zoom(...)创建多个zoom控件。

/**
* 设置map 默认zoomIn/zoomOut控件
*/
new Z.Map('mapObj',{
    center:  new Z.Coordinate(121.489935,31.24432), //初始中心点坐标
    zoomLevel:   11, //初始缩放级别
    zoomControl :  true, //缩放控件
    zoomControlOptions : { //设置缩放控件的位置
        position : Z.Control['top_left']//top_left|top_right|bottom_left|bottom_right
    }
});
/**
* 创建多个zoom控件
*/
var zoomControl = new Z.Control.Zoom({
    id: 'zoomControl1', //控件id
    //自定义控件放置位置
    position : {
        top: '40',
        right: '60'
    }
});
mapObj.addControl(zoomControl);

var zoomControl2 = new Z.Control.Zoom({
    id: 'zoomControl12', //控件id
    position : Z.Control['bottom_right']
});
mapObj.addControl(zoomControl2);

//构建panel
var panel = new Z.Panel({
    id: 'panelId', //panel id
    position : {//放置panel的位置
        top: '15',
        left: '15'
    },
    style: 'default',
    draggable: true,//能否拖动
    title: '地图信息面板',
    html: true, //content值能否为html
    content: '地图信息面板'
});
/**
* 将面板绑定到地图容器上
*/
mapObj.addControl(panel);
//创建图层
var layer = new Z.VectorLayer('layer');
//将图层添加到地图对象
mapObj.addLayer(layer);
//创建点的坐标对象
var markerCoordinate = new Z.Coordinate(121.489935,31.24432);
//创建点对象
var marker = new Z.Marker(markerCoordinate,'marker_id');
layer.addGeometry(marker);

var markerPanelOptions = {
    id : 'label01',
    position : {
        top: '15',
        right: '15'
    },
    style: 'default',
    draggable: true,
    title: '点信息面板',
    html: true,
    content: '点信息正文',
    linksymbol: {//panel与marker之间连线样式
        'line-color' : '#00ff00',
        'line-width' : 2,
        'line-opacity' : 1,
        'line-dasharray' : '-'
    }
};
var markerInfoPanel = new Z.Panel(markerPanelOptions);
marker.addPanel(markerInfoPanel);
